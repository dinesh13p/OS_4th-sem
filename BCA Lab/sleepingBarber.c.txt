#include <stdio.h>
#include <pthread.h>
#include <semaphore.h>
#include <unistd.h>

#define CHAIRS 3   // number of waiting chairs

sem_t customers;   // count of waiting customers
sem_t barber;      // barber available
sem_t mutex;       // for mutual exclusion
int waiting = 0;   // number of waiting customers

void* barber_fun(void* arg) {
    while (1) {
        sem_wait(&customers);          // wait for a customer
        sem_wait(&mutex);              // lock waiting room
        waiting--;                     // one customer goes for haircut
        printf("Barber is cutting hair. Waiting customers: %d\n", waiting);
        sem_post(&mutex);              // unlock waiting room
        sem_post(&barber);             // barber is busy
        sleep(2);                      // haircut time
    }
}

void* customer_fun(void* arg) {
    sem_wait(&mutex);                  // lock waiting room
    if (waiting < CHAIRS) {
        waiting++;
        printf("Customer entered shop. Waiting customers: %d\n", waiting);
        sem_post(&customers);          // signal barber
        sem_post(&mutex);              // unlock waiting room
        sem_wait(&barber);             // wait for barber
        printf("Customer is getting a haircut.\n");
    } else {
        printf("No chair available. Customer left.\n");
        sem_post(&mutex);              // unlock waiting room
    }
    return NULL;
}

int main() {
    pthread_t bthread, cthread[10];
    int i;

    sem_init(&customers, 0, 0);
    sem_init(&barber, 0, 0);
    sem_init(&mutex, 0, 1);

    pthread_create(&bthread, NULL, barber_fun, NULL);

    for (i = 0; i < 10; i++) {
        sleep(1);   // new customer arrives every 1 second
        pthread_create(&cthread[i], NULL, customer_fun, NULL);
    }

    pthread_join(bthread, NULL);
    for (i = 0; i < 10; i++) {
        pthread_join(cthread[i], NULL);
    }

    sem_destroy(&customers);
    sem_destroy(&barber);
    sem_destroy(&mutex);

    return 0;
}
