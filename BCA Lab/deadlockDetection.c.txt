#include <stdio.h>

int main() {
    int n = 5, m = 3, i, j, k;

    // Sample Allocation Matrix
    int alloc[5][3] = {
        {0, 1, 0},  // P0
        {2, 0, 0},  // P1
        {3, 0, 2},  // P2
        {2, 1, 1},  // P3
        {0, 0, 2}   // P4
    };

    // Sample Request Matrix
    int request[5][3] = {
        {0, 0, 0},  // P0
        {2, 0, 2},  // P1
        {0, 0, 0},  // P2
        {1, 0, 0},  // P3
        {0, 0, 2}   // P4
    };

    // Available resources
    int avail[3] = {0, 0, 0};

    int finish[5] = {0}, safe[5], safe_index = 0;
    int count = 0;

    // Deadlock Detection Algorithm
    while(count < n) {
        int found = 0;
        for(i = 0; i < n; i++) {
            if(finish[i] == 0) {
                int can_allocate = 1;
                for(j = 0; j < m; j++) {
                    if(request[i][j] > avail[j]) {
                        can_allocate = 0;
                        break;
                    }
                }
                if(can_allocate) {
                    for(k = 0; k < m; k++) {
                        avail[k] += alloc[i][k]; // Release resources
                    }
                    finish[i] = 1;
                    safe[safe_index++] = i;
                    found = 1;
                    count++;
                }
            }
        }
        if(!found) break; // No process can proceed, possible deadlock
    }

    // Check for deadlock
    int deadlock = 0;
    printf("\nProcesses in deadlock (if any): ");
    for(i = 0; i < n; i++) {
        if(finish[i] == 0) {
            printf("P%d ", i);
            deadlock = 1;
        }
    }

    if(!deadlock)
        printf(" No deadlock detected.\n");
    else
        printf("\n");

    return 0;
}
